# slurm-configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: slurm-config
data:
  slurm.conf: |
    ClusterName=kubernetes
    SlurmctldHost=slurmctld
    
    # COMPUTE NODES
    NodeName=compute[1-3] CPUs=4 RealMemory=4096 State=UNKNOWN
    PartitionName=normal Default=yes Nodes=compute[1-3] State=UP MaxTime=INFINITE
    
    # SCHEDULING
    SelectType=select/cons_tres
    SelectTypeParameters=CR_Core
    
    # LOGGING
    SlurmctldDebug=debug5
    SlurmdDebug=debug5
    SlurmctldLogFile=/var/log/slurm/slurmctld.log
    SlurmdLogFile=/var/log/slurm/slurmd.log
    
    # ACCOUNTING
    AccountingStorageType=accounting_storage/slurmdbd
    AccountingStorageHost=slurmdbd
    AccountingStoragePort=6819

---

# slurmdbd-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: slurmdbd
spec:
  replicas: 1
  selector:
    matchLabels:
      app: slurmdbd
  template:
    metadata:
      labels:
        app: slurmdbd
    spec:
      containers:
      - name: slurmdbd
        image: giovtorres/slurm-docker-cluster:latest
        command: ["/bin/bash", "-c", "sleep 30 && /usr/sbin/slurmdbd -D -vvv"]
        ports:
        - containerPort: 6819
        volumeMounts:
        - name: slurm-config
          mountPath: /etc/slurm/slurm.conf
          subPath: slurm.conf
      volumes:
      - name: slurm-config
        configMap:
          name: slurm-config

---

# slurmd-daemonset.yaml

  selector:
    matchLabels:
      app: slurmd
  template:
    metadata:
      labels:
        app: slurmd
    spec:
      containers:
      - name: slurmd

        command: ["/bin/bash", "-c", "sleep 30 && /usr/sbin/slurmd -D -vvv"]

          mountPath: /etc/slurm/slurm.conf

  - Egress

  - from:
    - podSelector:
        matchLabels:
          app: slurmd

      port: 6817
  egress:

    - podSelector:
        matchLabels:
